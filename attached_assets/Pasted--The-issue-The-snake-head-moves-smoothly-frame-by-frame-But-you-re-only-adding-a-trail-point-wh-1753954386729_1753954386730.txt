🔍 The issue:
The snake head moves smoothly frame-by-frame.

But you're only adding a trail point when distance exceeds SEGMENT_SPACING.

So the body segments "jump" in and don’t align with the head’s velocity — especially if speed and spacing aren’t tuned properly.

✅ The Real Slither.io-style Fix: Interpolated Trail Update
🔧 Fix strategy:
Use a continuous trail of head positions (record every frame).

Space body segments along that trail using distance, not frame count.

Always draw the head at the real-time position, and body segments behind it at correct intervals.

const SEGMENT_SPACING = 20;
const SEGMENT_RADIUS = 10;
const MASS_PER_SEGMENT = 1;

let totalMass = 6; // Starting length = 6 segments
let segmentTrail = []; // ALL head positions (one per frame)
let visibleSegments = [];

let head = { x: canvas.width / 2, y: canvas.height / 2 };

function updateSnake() {
  // Move head
  const dx = Math.cos(currentAngle) * speed;
  const dy = Math.sin(currentAngle) * speed;
  head.x += dx;
  head.y += dy;

  // Add head to trail every frame
  segmentTrail.unshift({ x: head.x, y: head.y });

  // Remove excess trail length (keep enough to render full snake)
  const maxTrailLength = Math.floor((totalMass / MASS_PER_SEGMENT) * SEGMENT_SPACING * 2);
  if (segmentTrail.length > maxTrailLength) {
    segmentTrail.length = maxTrailLength;
  }

  // Sample segments at fixed spacing from the trail
  visibleSegments = [];
  let distanceSoFar = 0;
  for (let i = 0; i < segmentTrail.length && visibleSegments.length < totalMass; i++) {
    const point = segmentTrail[i];
    if (distanceSoFar >= visibleSegments.length * SEGMENT_SPACING) {
      visibleSegments.push(point);
    }

    if (i < segmentTrail.length - 1) {
      const dx = segmentTrail[i].x - segmentTrail[i + 1].x;
      const dy = segmentTrail[i].y - segmentTrail[i + 1].y;
      distanceSoFar += Math.sqrt(dx * dx + dy * dy);
    }
  }
}

function drawSnake() {
  for (let i = visibleSegments.length - 1; i > 0; i--) {
    const seg = visibleSegments[i];
    const gradient = ctx.createRadialGradient(seg.x, seg.y, 0, seg.x, seg.y, SEGMENT_RADIUS);
    gradient.addColorStop(0, "#ffbaba");
    gradient.addColorStop(1, "#d55400");

    ctx.fillStyle = gradient;
    ctx.beginPath();
    ctx.arc(seg.x, seg.y, SEGMENT_RADIUS, 0, 2 * Math.PI);
    ctx.fill();
  }

  // Head (index 0)
  const headSeg = visibleSegments[0] || head;
  const headGradient = ctx.createRadialGradient(headSeg.x, headSeg.y, 0, headSeg.x, headSeg.y, SEGMENT_RADIUS);
  headGradient.addColorStop(0, "#ffffff");
  headGradient.addColorStop(1, "#d55400");

  ctx.fillStyle = headGradient;
  ctx.beginPath();
  ctx.arc(headSeg.x, headSeg.y, SEGMENT_RADIUS, 0, 2 * Math.PI);
  ctx.fill();

  drawEyes(headSeg.x, headSeg.y, currentAngle);
}
